from pyldt import LdtWriter
from ldt_symmetry import LdtAutoDetector, LdtSymmetriser

# ── automatic detection ───────────────────────────────────────────────
detector = LdtAutoDetector()
isym = detector.detect(ldt)

# ── symmetrise ────────────────────────────────────────────────────────
if isym != 0:
    ldt_sym = LdtSymmetriser.symmetrise(ldt, isym=isym)
    LdtWriter.write(ldt_sym, f"{h.luminaire_name}_SYM.ldt")
    eulumdat = f"{h.luminaire_name}_SYM.ldt"
    svg = plot_ldt(eulumdat, interpolate=True, interp_method="cubic")
else:
    print("No plausible symmetry detected — file left unchanged.")
